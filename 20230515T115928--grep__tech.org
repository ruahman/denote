* grep -Rin "pattern" /path/to/folder
  search folder where pattern occurs and show line number

  - R :: recursive
  - i :: ignore
  - n :: show line number
  
* grep -Ril "pattern" /path/to/folder
  search folder where patern occurs but list only the files

  - R :: recursive
  - i :: ignore
  - l :: list file
