\section{circle equation}
\label{sec:orgae471af}

\begin{align*}
radius &= \sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2}\\
{radius}^2 &= (\sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2})^2\\
{radius}^2 &= (x_2 - x_1)^2 + (y_2 - y_1)^2
\end{align*}

$$
(x_2 - x_1)^2 + (y_2 - y_1)^2 = {radius}^2
$$
